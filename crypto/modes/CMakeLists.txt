    set(
        MODES_ARCH_SOURCES

        ghash-x86_64.${ASM_EXT}
    )

if (${ARCH} STREQUAL "x86")
    set(
        MODES_ARCH_SOURCES

    )
endif()

    set(
        MODES_ARCH_SOURCES

        ghash-armv4.${ASM_EXT}
    )
endif()

    cfb128.c
    ctr128.c
    cts128.c
    gcm128.c
    ofb128.c
    xts128.c

    ${MODES_ARCH_SOURCES}
